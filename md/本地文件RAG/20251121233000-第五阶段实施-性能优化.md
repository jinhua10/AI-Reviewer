@Slf4j
public class ConcurrentBatchIndexer {
    
    private final ExecutorService executor;
    private final int batchSize;
    
    public ConcurrentBatchIndexer(int threads, int batchSize) {
        this.executor = Executors.newFixedThreadPool(threads);
        this.batchSize = batchSize;
    }
    
    /**
     * å¹¶å‘æ‰¹é‡ç´¢å¼•æ–‡æ¡£
     */
    public void indexConcurrently(List<Document> documents) throws InterruptedException {
        List<Future<?>> futures = new ArrayList<>();
        
        // åˆ†æ‰¹å¤„ç†
        for (int i = 0; i < documents.size(); i += batchSize) {
            int end = Math.min(i + batchSize, documents.size());

            Future<?> future = executor.submit(() -> {
                try {
                    indexBatch(batch);
                } catch (Exception e) {
                    log.error("Failed to index batch", e);
                }
            });
            
            futures.add(future);
        }
        
        // ç­‰å¾…æ‰€æœ‰ä»»åŠ¡å®Œæˆ
        for (Future<?> future : futures) {
            try {
                future.get();
            } catch (ExecutionException e) {
                log.error("Index task failed", e);
            }
        }
        
        log.info("Concurrent indexing completed for {} documents", documents.size());
    }
    
    private void indexBatch(List<Document> batch) {
        // å®é™…çš„æ‰¹é‡ç´¢å¼•é€»è¾‘
        log.debug("Indexing batch of {} documents", batch.size());
    }

@Slf4j
public class BatchProcessor<T> {
    
    private final int batchSize;
    private final Consumer<List<T>> processor;
    private final List<T> buffer;
    
    public BatchProcessor(int batchSize, Consumer<List<T>> processor) {
        this.batchSize = batchSize;
        this.processor = processor;
        this.buffer = new ArrayList<>(batchSize);
    }
    
    /**
     * æ·»åŠ å…ƒç´ ï¼Œè‡ªåŠ¨æ‰¹é‡å¤„ç†
     */
    public void add(T item) {
        buffer.add(item);

            flush();
        }
    }
    
    /**
     * åˆ·æ–°ç¼“å†²åŒº
     */
    public void flush() {
        if (!buffer.isEmpty()) {
            try {
                processor.accept(new ArrayList<>(buffer));
                buffer.clear();
            } catch (Exception e) {
                log.error("Batch processing failed", e);
            }
        }
    }
    
    /**
     * æ‰¹é‡å¤„ç†åˆ—è¡¨
     */
    public static <T> void processList(List<T> items, int batchSize, 
                                        Consumer<List<T>> processor) {
        for (int i = 0; i < items.size(); i += batchSize) {
            int end = Math.min(i + batchSize, items.size());
            List<T> batch = items.subList(i, end);
            processor.accept(batch);
        }
    }
